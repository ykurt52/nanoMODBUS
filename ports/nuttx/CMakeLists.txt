if(CONFIG_NANOMODBUS)

  set(CSRCS nanomodbus.c)
  set(CFLAGS)

  if(NOT CONFIG_NANOMODBUS_SERVER)
    list(APPEND CFLAGS -DNMBS_SERVER_DISABLED)
  else()
    if(NOT CONFIG_NANOMODBUS_SERVER_READ_COILS)
      list(APPEND CFLAGS -DNMBS_SERVER_READ_COILS_DISABLED)
    endif()
    if(NOT CONFIG_NANOMODBUS_SERVER_READ_DISCRETE_INPUTS)
      list(APPEND CFLAGS -DNMBS_SERVER_READ_DISCRETE_INPUTS_DISABLED)
    endif()
    if(NOT CONFIG_NANOMODBUS_SERVER_READ_HOLDING_REGISTERS)
      list(APPEND CFLAGS -DNMBS_SERVER_READ_HOLDING_REGISTERS_DISABLED)
    endif()
    if(NOT CONFIG_NANOMODBUS_SERVER_READ_INPUT_REGISTERS)
      list(APPEND CFLAGS -DNMBS_SERVER_READ_INPUT_REGISTERS_DISABLED)
    endif()
    if(NOT CONFIG_NANOMODBUS_SERVER_WRITE_SINGLE_COIL)
      list(APPEND CFLAGS -DNMBS_SERVER_WRITE_SINGLE_COIL_DISABLED)
    endif()
    if(NOT CONFIG_NANOMODBUS_SERVER_WRITE_SINGLE_REGISTER)
      list(APPEND CFLAGS -DNMBS_SERVER_WRITE_SINGLE_REGISTER_DISABLED)
    endif()
    if(NOT CONFIG_NANOMODBUS_SERVER_WRITE_MULTIPLE_COILS)
      list(APPEND CFLAGS -DNMBS_SERVER_WRITE_MULTIPLE_COILS_DISABLED)
    endif()
    if(NOT CONFIG_NANOMODBUS_SERVER_WRITE_MULTIPLE_REGISTERS)
      list(APPEND CFLAGS -DNMBS_SERVER_WRITE_MULTIPLE_REGISTERS_DISABLED)
    endif()
    if(NOT CONFIG_NANOMODBUS_SERVER_READ_WRITE_REGISTERS)
      list(APPEND CFLAGS -DNMBS_SERVER_READ_WRITE_REGISTERS_DISABLED)
    endif()
    if(NOT CONFIG_NANOMODBUS_SERVER_READ_FILE_RECORD)
      list(APPEND CFLAGS -DNMBS_SERVER_READ_FILE_RECORD_DISABLED)
    endif()
    if(NOT CONFIG_NANOMODBUS_SERVER_WRITE_FILE_RECORD)
      list(APPEND CFLAGS -DNMBS_SERVER_WRITE_FILE_RECORD_DISABLED)
    endif()
    if(NOT CONFIG_NANOMODBUS_SERVER_READ_DEVICE_IDENTIFICATION)
      list(APPEND CFLAGS -DNMBS_SERVER_READ_DEVICE_IDENTIFICATION_DISABLED)
    endif()
  endif()

  if(NOT CONFIG_NANOMODBUS_CLIENT)
    list(APPEND CFLAGS -DNMBS_CLIENT_DISABLED)
  endif()

  if(NOT CONFIG_NANOMODBUS_STRERROR)
    list(APPEND CFLAGS -DNMBS_STRERROR_DISABLED)
  endif()

  target_sources(apps PRIVATE ${CSRCS})

  if(CFLAGS)
    set_source_files_properties(${CSRCS} PROPERTIES COMPILE_FLAGS
                                                     "${CFLAGS}")
  endif()

endif()
